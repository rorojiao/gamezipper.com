### Core Algorithms

#### Validation (Check if move is valid)
```javascript
function isValid(board, row, col, num) {
  // Check row
  for (let c = 0; c < 9; c++) {
    if (board[row][c] === num) return false;
  }
  // Check column
  for (let r = 0; r < 9; r++) {
    if (board[r][col] === num) return false;
  }
  // Check 3×3 box
  const boxRow = Math.floor(row / 3) * 3;
  const boxCol = Math.floor(col / 3) * 3;
  for (let r = boxRow; r < boxRow + 3; r++) {
    for (let c = boxCol; c < boxCol + 3; c++) {
      if (board[r][c] === num) return false;
    }
  }
  return true;
}
```

#### Candidate Suggestion (Smart Notes)
```javascript
function getCandidates(board, row, col) {
  if (board[row][col] !== 0) return [];
  const candidates = [];
  for (let num = 1; num <= 9; num++) {
    if (isValid(board, row, col, num)) {
      candidates.push(num);
    }
  }
  return candidates;
}

function updateAllNotes(board, notes) {
  for (let r = 0; r < 9; r++) {
    for (let c = 0; c < 9; c++) {
      if (board[r][c] === 0) {
        notes[r][c] = getCandidates(board, r, c).reduce((mask, n) => mask | (1 << (n-1)), 0);
      } else {
        notes[r][c] = 0;
      }
    }
  }
}
```

#### Solver (Backtracking with Constraint Propagation)
```javascript
function solve(board, useProp = true) {
  // Find empty cell with fewest candidates (MRV heuristic)
  let minCandidates = 10;
  let bestCell = null;

  for (let r = 0; r < 9; r++) {
    for (let c = 0; c < 9; c++) {
      if (board[r][c] === 0) {
        const candidates = getCandidates(board, r, c);
        if (candidates.length < minCandidates) {
          minCandidates = candidates.length;
          bestCell = {r, c, candidates};
          if (minCandidates === 0) return null; // Dead end
        }
      }
    }
  }

  if (!bestCell) return board; // Solved

  const {r, c, candidates} = bestCell;
  for (const num of candidates) {
    const newBoard = board.map(row => [...row]);
    newBoard[r][c] = num;
    const result = solve(newBoard, false);
    if (result) return result;
  }

  return null; // No solution
}
```

#### Generator (Create Puzzle from Solution)
```javascript
function generate(difficulty) {
  // 1. Generate complete valid board
  const board = Array(9).fill(null).map(() => Array(9).fill(0));
  fillBoard(board); // Randomly fill diagonal boxes, then solve

  // 2. Remove cells based on difficulty
  const targetGivens = {
    beginner: 50, easy: 45, medium: 36, hard: 30, expert: 25, master: 22
  }[difficulty] || 36;

  let cellsToRemove = 81 - targetGivens;
  const puzzle = board.map(row => [...row]);

  while (cellsToRemove > 0) {
    const r = Math.floor(Math.random() * 9);
    const c = Math.floor(Math.random() * 9);

    if (puzzle[r][c] !== 0) {
      const backup = puzzle[r][c];
      puzzle[r][c] = 0;

      // Verify unique solution
      const solutions = countSolutions(puzzle, 2);
      if (solutions === 1) {
        cellsToRemove--;
      } else {
        puzzle[r][c] = backup; // Restore
      }
    }
  }

  return puzzle;
}

function countSolutions(board, maxSolutions) {
  const solutions = [];
  function solveRec(b) {
    if (solutions.length >= maxSolutions) return;

    const empty = findEmpty(b);
    if (!empty) {
      solutions.push(b.map(row => [...row]));
      return;
    }

    const {r, c} = empty;
    for (let num = 1; num <= 9; num++) {
      if (isValid(b, r, c, num)) {
        b[r][c] = num;
        solveRec(b);
        b[r][c] = 0;
      }
    }
  }

  solveRec(board.map(row => [...row]));
  return solutions.length;
}
```
